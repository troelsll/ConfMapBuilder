ğŸ§± Database Schema (Prisma)
model Basemap {
  id          String   @id @default(cuid())
  name        String
  description String?
  imagePath   String
  widthPx     Int
  heightPx    Int
  poiPlacements PoiPlacement[]
  createdAt   DateTime @default(now())
}

model Category {
  id    String @id @default(cuid())
  name  String @unique
  color String?
  pois  POI[]
}

model POI {
  id          String   @id @default(cuid())
  name        String
  description String?
  icon        String?
  categoryId  String
  category    Category  @relation(fields: [categoryId], references: [id])
  placements  PoiPlacement[]
}

model PoiPlacement {
  id        String  @id @default(cuid())
  basemapId String
  poiId     String
  x         Float
  y         Float
  basemap   Basemap @relation(fields: [basemapId], references: [id])
  poi       POI     @relation(fields: [poiId], references: [id])
}

model EventMap {
  id        String   @id @default(cuid())
  name      String
  dateRange String
  location  String
  basemapId String
  basemap   Basemap  @relation(fields: [basemapId], references: [id])
  sections  Section[]
  pdfPath   String?
  createdAt DateTime  @default(now())
}

model Section {
  id         String   @id @default(cuid())
  title      String
  eventMapId String
  eventMap   EventMap @relation(fields: [eventMapId], references: [id])
  items      SectionItem[]
}

model SectionItem {
  id         String  @id @default(cuid())
  sectionId  String
  poiId      String
  note       String?
  section    Section @relation(fields: [sectionId], references: [id])
  poi        POI     @relation(fields: [poiId], references: [id])
}
